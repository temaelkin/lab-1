#include <math.h>

{
    double x, y;
    double pi = 3.1415;
    double T1 = 687;
    double T2 = 365;
    double r1 = 228; //10^6 km

        printf("In current moment t = %d\n", t);
        x = r1 * cos(w1*t) - r2 * cos(w2*t);

        printf("Y coordinate is %.5f\n", y);

}

| a              | int        | Массив                  |
### Код программы
```c
#include <stdio.h>

int main()
{
    int i, j, key;
    int a[5] = {5, 2, 7, 1, 3};
    printf("Unsorted array:\n");

    for (i = 0; i < 5; i++)
        printf("%d ", a[i]);

    printf("\n");

    for (i = 1; i < 5; i++)
    {
        key = a[i];
        j = i - 1;
        while (j >= 0 && a[j] > key)
        {
            a[j + 1] = a[j];
            j--;
        }
        a[j + 1] = key;
    }

    printf("Sorted array:\n");

    for (i = 0; i < 5; i++)
        printf("%d ", a[i]);

    return 0;
}
```
